private val clients = mutableSetOf<WebSocket>()

// Update of the WebSocket server
override fun onOpen(conn: WebSocket, handshake: ClientHandshake) {
    Log.d("WebSocketServer", "Client connected")

    synchronized(clients) {
        clients.add(conn)
    }
}

override fun onClose(conn: WebSocket, code: Int, reason: String, remote: Boolean) {

    synchronized(clients) {
        clients.remove(conn)
    }

    Log.d("WebSocketServer", "Client disconnected")
}

// Send to evrey client
if (now - lastFrameSent >= frameInterval) {

    try {
        val stream = ByteArrayOutputStream()
        bitmap.compress(Bitmap.CompressFormat.JPEG, jpegQuality, stream)

        val frame = stream.toByteArray()

        synchronized(clients) {
            clients.removeAll { !it.isOpen }

            clients.forEach { client ->
                client.send(frame)
            }
        }

        lastFrameSent = now

    } catch (e: Exception) {
        Log.e("DetectionActivity", "Error sending frame", e)
    }
}

import websocket
import numpy as np
import cv2

from ultralytics import YOLO

PHONE_IP = "192.168.X.X"

model = YOLO("yolo11n.pt")

ws = websocket.WebSocket()
ws.connect(f"ws://{PHONE_IP}:8080")

print("Connected!")

while True:

    frame = ws.recv()

    if not isinstance(frame, bytes):
        continue

    # JPEG bytes -> numpy array
    nparr = np.frombuffer(frame, np.uint8)

    # numpy -> OpenCV image
    image = cv2.imdecode(nparr, cv2.IMREAD_COLOR)

    if image is None:
        continue

    results = model(image, verbose=False)

    for result in results:
        for box in result.boxes:

            cls = int(box.cls)
            conf = float(box.conf)

            print(model.names[cls], round(conf, 2))
